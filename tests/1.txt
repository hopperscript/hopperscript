define var "test"
define object "idk" = objects.square
define object "more" = objects.square

define ability "hmm(os)" {
  move_forward("20")
}

for "idk" {
  when game_starts() {
    move_forward("900")
    set_color(#0dd)
    set_variable(v"idk"."test", "-100")
    move_forward(v"idk"."test")
  }

  when is_tapped(o"more") {
    ability!("hmm(os)")
    destroy()
  }

  when is_tapped(o"idk") {
    set_variable(v"idk"."test", "110")
  }

  when cond(v"idk"."test" != "-100") {
    destroy()
  }

  when cond((v"test" == "100")) {
    grow_by("100")
    increase_variable(v"test", "1")
  }
}

for "more" {
  when game_starts() {
    set_variable(v"test", "100")
    move_forward(v"test")
  }

  when is_tapped(o"idk") {
    move_forward("100")
  }

  when cond(v"test" == "100") {
    grow_by("100")
    increase_variable(v"test", "1")
  }

  when is_touching(o"more", o"idk") {
    move_forward("-100")
  }
}
