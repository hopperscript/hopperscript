define var "test"
define object "idk" = objects.Square
define object "more" = objects.Square

define ability "hmm(os)" {
  move_forward("20")

  ability!("hmm(os)")
}

for "idk" {
  when game_starts() {
    move_forward("900")
    
    set_variable(v"idk"."test", "-100")
    move_forward(v"idk"."test")
  }

  when is_tapped(o"more") {
    ability!("hmm(os)")
    destroy()
  }
}

for "more" {
  when game_starts() {
    set_variable(v"test", "100")
    move_forward(v"test")
  }

  when is_tapped(o"more") {
    turn("60")
    set_position(v"test", "90")
    flip()
    change_x(v"idk"."test")
    change_y(v"idk"."test")
    set_speed("-9999")
    set_angle("80")
    set_origin("10", v"more"."test")
  }
}
